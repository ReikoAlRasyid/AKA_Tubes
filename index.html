function luhnIterative(cardNumber) {
    let total = 0;
    let isEven = false;
    
    for (let i = cardNumber.length - 1; i >= 0; i--) {
        let digit = parseInt(cardNumber[i]);
        
        if (isEven) {
            digit *= 2;
            if (digit > 9) digit -= 9;
        }
        
        total += digit;
        isEven = !isEven;
    }
    
    return total % 10 === 0;
}
function luhnRecursive(cardNumber, 
                       index = 0, 
                       total = 0, 
                       isEven = false) {
    
    if (index >= cardNumber.length) {
        return total % 10 === 0;
    }
    
    const reverseIndex = cardNumber.length - index - 1;
    let digit = parseInt(cardNumber[reverseIndex]);
    
    if (isEven) {
        digit *= 2;
        if (digit > 9) digit -= 9;
    }
    
    return luhnRecursive(cardNumber,
                         index + 1,
                         total + digit,
                         !isEven);
}
